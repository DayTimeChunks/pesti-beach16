# Generation of clone.map
# 1. Go to Nutshell, click icon "make a new empty map"
# 2. Select nrows, and ncols (same as your ".asc" (raster) files)
# 3. Data type = Scalar
# 4. Projection = Y bottom to Top
# 5. Input corners, found also on "asc" file, open with ArcGis -> Layer Properties
# ArcGis -> Layer Properties -> Source -> Extent:
# X Upper Left corner = "Right" 
# Y Upper Left corner = "Top"
# Cell length = 2

# Convert ".asc" to ".map"

asc2map --clone clone.map -S -a landuse.asc landuse.map

# PCRCalc:
# https://blog.utwente.nl/lisem/running-lisem/maps/

# Create ldd.map

pcrcalc --lddin ldd.map = lddcreate(dem_ldd.map, 1E35, 1E35, 1E35, 1E35)
pcrcalc version: Apr 16 2015 (win64/msvc)

# Create outlet.map

pcrcalc outlet.map = pit(ldd.map)
pcrcalc version: Apr 16 2015 (win64/msvc)

pcrcalc area.map = catchment(ldd.map, outlet.map)
pcrcalc version: Apr 16 2015 (win64/msvc)



# Convert .map to asc:
map2asc outlet_true.map outlet_tru.asc
map2asc version: Apr 16 2015 (win64/msvc)
